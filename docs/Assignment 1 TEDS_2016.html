Clean the Data

To clean this data, the following is done:

(1) “labelled” variables that were stored as numbers with labels were converted to readable categorical factors.

(2) Non-responses were re-coded as missing (NA).

(3) Binary variables were forced to 0,1, or NA.

(4) Removed duplicate variables & renamed.

(5) Implausible negative values in variables like income were set to NA.

(6) Overlapping region variables were simplified, keeping one consistent measure.

library(haven)
library(labelled)
library(skimr)

# -------------------------------
# Preserve raw data
# -------------------------------
TEDS_2016_raw <- TEDS_2016
df <- TEDS_2016

# -------------------------------
# Convert labelled variables to factors
# -------------------------------
df <- df %>%
  mutate(across(where(is.labelled), as_factor))

# -------------------------------
# Recode true nonresponses to NA
# -------------------------------

# Ethnicity
df <- df %>%
  mutate(Ethnic = na_if(Ethnic, "Noresponse"))

# TONDU variables
df <- df %>%
  mutate(
    Tondu  = na_if(Tondu, "Nonresponse"),
    Tondu3 = na_if(Tondu3, "Nonresponse")
  )

# Political knowledge question
df <- df %>%
  mutate(
    nI2 = na_if(nI2, "Refuse to answer"),
    nI2 = na_if(nI2, "Don't know")
  )

# -------------------------------
# Clean binary variables (force 0/1 or NA)
# -------------------------------
binary_vars <- c(
  "votetsai", "votetsai_nm", "votetsai_all",
  "votekmt", "votekmt_nm",
  "voteblue", "voteblue_nm",
  "Blue", "Green", "No_Party",
  "Independence", "Unification",
  "sq", "Taiwanese",
  "KMT", "DPP", "npp", "pfp", "noparty",
  "Govt_for_public", "Govt_dont_care"
)

df <- df %>%
  mutate(across(all_of(binary_vars), ~ ifelse(. %in% c(0, 1), ., NA)))

# -------------------------------
# Resolve duplicated concepts
# -------------------------------

# Education: keep labelled version
df <- df %>% select(-edu)

# Age: keep both but rename
df <- df %>%
  rename(
    age_group = Age,
    age_years = age
  )

# -------------------------------
# Income cleanup
# -------------------------------
df <- df %>%
  mutate(
    income    = ifelse(income < 0, NA, income),
    income_nm = ifelse(income_nm < 0, NA, income_nm)
  )

# -------------------------------
# Geography redundancy
# -------------------------------
df <- df %>% select(-north, -South)
